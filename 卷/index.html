//校门外的树
#include <iostream>
using namespace std;
bool tree[10000];
void log(int Start,int End)
{
    for(int i = Start; i <= End; i++)
    {
        tree[i] = false;
    }
}
int main() 
{
    int l,m;
    int cnt = 0;
    cin >> l >> m;
    for(int i = 0; i <= l; i++)
    {
        tree[i] = true;
    }
    for(int i = 0; i < m; i++)
    {
        int u,v;
        cin >> u >> v;
        log(u,v);
    }
    for(int i = 0; i <= l; i++)
    {
        if(tree[i]) cnt++;
    }
    cout<<cnt;
    return 0;
}

//工艺品制作
#include <iostream>
using namespace std;
int craft[21][21][21]={0};
void cut(int x1,int y1,int z1,int x2,int y2,int z2)
{
    for(int i = x1; i <= x2; i++)
        for(int j = y1; j <= y2; j++)
            for(int k = z1; k <= z2; k++)
                craft[i][j][k] = 1;
}
int main() 
{
    int width,length,height;
    cin >> width >> length >> height;
    int cnt = 0;
    int q;
    cin>>q;
    for(int i = 0;i < q;i++)
    {
        int x1,y1,z1,x2,y2,z2;
        cin >> x1 >> y1 >> z1 >> x2 >> y2 >> z2;
        cut(x1,y1,z1,x2,y2,z2);
    }
    for(int i = 1; i <= width; i++)
        for(int j = 1; j <= length; j++)
            for(int k = 1; k <= height; k++)
            {
                if (craft[i][j][k]==0)
                {
                    cnt++;
                }
                
            }
    cout << cnt;
    return 0;
}

#include <iostream>
#include <cmath>
using namespace std;
int digit_get(int n)
{
    int ans=0;
    while (n)
    {
        n/=10;
        ans++;
    }
    return ans;
}
bool p1(int n)//chk palindromes(并尽可能的留下质数)，范围[5,100000000]
{
    if(n==5||n==7||n==11) return true;//特判
    
    else if(n % 2 == 0) return false; 
    else if(n % 3 == 0) return false;
    else if(n / 100 == 0) return false; //缩小范围
    int digit = digit_get(n);  //偶数位的回文数一定是11的倍数，统统哒咩
    if (digit == 3)      //ABA
    {
        if(n / 100 == n % 10) return true; 
        else return false;
    }
    else if (digit == 5) //ABCBA
    {
        if(n/10000 == n % 10 && (n/1000)%10 == (n/10)%10) return true;
        else return false;
    }
    else if (digit == 7) //ABCDCBA
    {
        if(n/1000000 == n % 10 && (n/100000)%10 == (n/10)%10 && (n/10000)%10 == (n/100)%10) return true;
        else return false;
    }
    else return false;
}
bool isp(int n) //chk prime
{
    if(n==2) return true;
    else if (n % 2==0||n == 1) return false;
    double sqrtn=sqrt(n);
    for(int i = 3; i < sqrtn + 1; i+=2)
        if(n % i== 0) return false;
    return true;
}
int main()
{
    int a,b;
    cin>>a>>b;
    for(int i = a;i <= b; i++)
        if(p1(i))
            if(isp(i))
                cout<<i<<endl;
    return 0;
}
